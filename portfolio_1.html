} }); Logger.log("✅ 股價更新完成 " + rows.length + " 檔 (" + now + ")"); } // ════════════════════════════════════════════════════════════ // Part 2：FIFO 計算 + 投資組合總覽（沿用你的原始邏輯） // ════════════════════════════════════════════════════════════ function updatePortfolio() { var ss = SpreadsheetApp.getActiveSpreadsheet(); } function toKey(d) { if (typeof d === "number") return d; if (d instanceof Date) return d.getTime(); return new Date(d).getTime(); } txRows.sort(function(a, b) { return toKey(a.date) - toKey(b.date); }); // FIFO var queue = {}, realized = {}; txRows.forEach(function(tx) { var t = tx.ticker; if (!queue[t]) queue[t] = []; if (!realized[t]) realized[t] = { netPnL: 0, soldShares: 0 }; if (tx.action === "買進") { queue[t].push({ shares: tx.shares, costPerShare: (tx.price * tx.shares + tx.fee) / tx.shares, buyFeePerShare: tx.fee / tx.shares }); realized[t].netPnL += netPnL; realized[t].soldShares += tx.shares; } }); // 整理持倉 var holdings = []; for (var tk in queue) { var lots = queue[tk], totalShares = 0, totalCost = 0; lots.forEach(function(l) { totalShares += l.shares; totalCost += l.shares * l.costPerShare; }); if (totalShares <= 0) continue; holdings.push({ ticker: tk, shares: totalShares, totalCost: totalCost }); } // 先更新 prices 分頁取得最新股價 updatePrices(); // 從 prices 分頁讀價格 var priceMap = getPricesFromSheet(ss); var unreal = mktVal - h.totalCost; var real = realized[h.ticker] ? realized[h.ticker].netPnL : 0; totalMktVal += mktVal; totalUnreal += unreal; totalRealized += real; holdingRows.push({ ticker: h.ticker, shares: h.shares, avgCost: r2(avgCost), price: price, mktVal: r2(mktVal), unreal: r2(unreal), unrealPct: h.totalCost > 0 ? unreal / h.totalCost : 0, real: r2(real) }); } // 從 prices 分頁讀取股價（A=代號 C=現價） function getPricesFromSheet(ss) { var sheet = ss.getSheetByName("prices"); if (!sheet) return {}; var data = sheet.getDataRange().getValues(); var map = {}; for (var i = 1;
